# ============================================================================
# Test Patterns for Space and Tab Character Handling
# ============================================================================
#
# These patterns specifically test that space and tab characters are properly
# handled in the NFA/DFA construction. This test group exists to catch regressions
# where spaces might be skipped instead of creating proper transitions.
#
# Expected behavior:
# - Commands with spaces like "git status" should match the full pattern
# - The space character (ASCII 32) should create transitions on Symbol 2
# - The tab character (ASCII 9) should create transitions on Symbol 3
# - Multiple consecutive spaces should create multiple space transitions
# - Mixed spaces and tabs should both work
# ============================================================================

# Longer patterns first (to avoid early matching)
[safe:space:test] cmd arg1 arg2 arg3
[safe:space:test] cmd arg1 arg2
[safe:space:test] CMD VAR1 VAR2
[safe:space:test] XYZ XYZ
[safe:space:test] cmd arg1
[safe:space:test] CMD VAR
[safe:space:test] XYZ
[caution:space:cmd_only] cmd

# Multiple spaces
[safe:space:test] cmd  arg1
[safe:space:test] cmd   arg1

# Commands that should NOT match (wrong spacing)
# These are negative test cases to ensure we don't over-match

# End of test patterns
