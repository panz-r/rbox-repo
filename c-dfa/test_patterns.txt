# Test patterns for enhanced whitespace handling

# Normalizing space tests (should match any sequence of spaces/tabs)
[safe:file:read] ls * -> allow
[safe:file:read] cat file -> allow
[safe:file:read] grep pattern file -> allow

# Verbatim space tests (should match exactly)
[safe:file:read] ls' 'file -> allow
[safe:file:read] cat' 'file -> allow

# Escaped character tests
[safe:file:read] echo\'test\' -> allow
[safe:file:read] file\ with\ spaces -> allow

# Special sequence tests
[safe:file:read] echo\ttab -> allow
[safe:file:read] echo\nnewline -> allow

# Mixed tests
[safe:file:read] command' 'with\'quotes\' -> allow
