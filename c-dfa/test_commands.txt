# ReadOnlyBox Test Commands
# Comprehensive test cases for DFA validation

# Fragment definitions for reuse
[fragment:SAFE::FILENAME] [a-zA-Z0-9_./-]

# Safe file operations (read-only)
[safe:file:read] cat ((SAFE::FILENAME))+
[safe:file:read] cat *.txt
[safe:file:read] cat file1.txt file2.txt
[safe:file:read] less file.txt
[safe:file:read] head file.txt
[safe:file:read] tail file.txt
[safe:file:read] grep pattern file.txt
[safe:file:read] grep -r pattern *.txt
[safe:file:read] find . -name "*.txt"
[safe:file:read] find /path -name "*.log"
[safe:file:read] ls -la
[safe:file:read] ls *.txt
[safe:file:read] pwd
[safe:file:read] stat file.txt
[safe:file:read] file file.txt
[safe:file:read] wc -l file.txt
[safe:file:read] du -sh .
[safe:file:read] df -h

# Modifying operations (should be blocked)
[modifying:file:write] rm file.txt
[modifying:file:write] rm -rf /
[modifying:file:write] touch file.txt
[modifying:file:write] mkdir dir
[modifying:file:write] rmdir dir
[modifying:file:write] mv file1.txt file2.txt
[modifying:file:write] cp file1.txt file2.txt
[modifying:file:write] chmod 755 file.txt
[modifying:file:write] chown user:group file.txt
[modifying:file:write] ln -s target link
[modifying:file:write] echo "text" > file.txt
[modifying:file:write] echo "text" >> file.txt

# Dangerous operations (should be blocked)
[dangerous:system:execute] reboot
[dangerous:system:execute] shutdown
[dangerous:system:execute] halt

[dangerous:system:execute] sudo rm -rf /

# Complex command patterns with pipes and redirects
[safe:file:read] cat file.txt | grep pattern
[safe:file:read] cat file.txt | head -10
[safe:file:read] cat file.txt | tail -5
[safe:file:read] grep pattern file.txt | wc -l

[safe:file:read] git log --oneline | head -10
[safe:file:read] ps aux | grep process

# Test patterns with special characters and escaping
[safe:file:read] cat file\ with\ spaces.txt
[safe:file:read] grep "pattern with spaces" file.txt
[safe:file:read] find . -name "file\*.txt"
[safe:file:read] echo "test with quotes"

# Test patterns with wildcards
[safe:file:read] ls *.txt
[safe:file:read] ls *.log
[safe:file:read] find . -name "*.txt"
[safe:file:read] grep pattern *

# Test patterns with options and flags
[safe:file:read] cat -n file.txt
[safe:file:read] grep -i pattern file.txt
[safe:file:read] grep -r --include="*.txt" pattern .
[safe:file:read] find . -type f -name "*.txt"
[safe:file:read] ls -la -h
[safe:file:read] ps aux --sort=-%mem
