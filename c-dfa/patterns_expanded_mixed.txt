[fragment:y] y
[fragment:c] c
[fragment:d] d

# Literal followed by fragment with quantifier
[safe:mixed] x((y))

[safe:mixed] ((x))y

# Multiple literals with fragment

# Fragment between literals
[safe:mixed] start((mid))+end
