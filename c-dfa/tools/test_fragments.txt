# Test patterns with fragment references

# Define fragments with scoped names (using :: syntax)
# Format: SCOPE::NAME = pattern

SAFE::FILENAME = [a-zA-Z0-9_.]+
SAFE::PATHNAME = (/[a-zA-Z0-9_.]+)+
SAFE::OPTIONS = [-][a-z]+
SAFE::GIT_CMD = log|show|diff|status|blame|branch|tag|remote|fetch|ls-files|rev-parse

# Use fragments in patterns with quantifiers
[safe] git ((SAFE::GIT_CMD)) ((SAFE::OPTIONS))*
[safe] cat ((SAFE::FILENAME))
[safe] ls ((SAFE::PATHNAME))
[safe] find ((SAFE::PATHNAME)) -name ((SAFE::FILENAME))
