# ReadOnlyBox Command Specification
# Format: [category] command_pattern
# Categories: safe, caution, modifying, dangerous, network, admin
# Note: DFA is a fast-path filter. Commands not matched go to server for full validation.

# Safe read-only commands (simple commands with no complex arguments)
[safe] cat
[safe] cat *
[safe] head
[safe] head *
[safe] tail
[safe] tail *
[safe] grep
[safe] grep * *
[safe] less
[safe] less *
[safe] more
[safe] more *
[safe] pager
[safe] pager *
[safe] ls
[safe] ls *
[safe] wc
[safe] wc *
[safe] ps
[safe] ps *
[safe] df
[safe] df *
[safe] du
[safe] du *
[safe] uname
[safe] uname *

# Git read-only commands (base command only - arguments validated by server)
[safe] git log
[safe] git show
[safe] git diff
[safe] git status
[safe] git blame
[safe] git blame *
[safe] git branch
[safe] git branch *
[safe] git tag
[safe] git tag *
[safe] git remote
[safe] git remote *
[safe] git fetch
[safe] git fetch *
[safe] git ls-files
[safe] git ls-files *
[safe] git rev-parse
[safe] git rev-parse *

# Note: 'find' commands are NOT in the DFA because they require detailed
# argument validation to detect dangerous options like -exec, -delete, etc.
# All 'find' commands go to the server where rofind module does full validation.
