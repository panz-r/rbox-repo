# ReadOnlyBox Command Specification
# Format: [category] command_pattern
# Categories: safe, caution, modifying, dangerous, network, admin
# Note: DFA is a fast-path filter. Commands not matched go to server for full validation.

# Safe read-only commands (simple commands with no complex arguments)
# Include both basename and full path versions
[safe] cat
[safe] cat *
[safe] /usr/bin/cat
[safe] /usr/bin/cat *

# ============ FILENAME FRAGMENT ============
# Fragment for matching filenames (used in capture patterns)
[fragment:FILENAME] [a-zA-Z0-9_./-]+

# ============ CAT WITH CAPTURE PATTERNS ============
# These patterns capture the filename argument for later analysis
# Use <tagname>...</tagname> syntax for capture tags
[safe:cat:capture] cat <filename>((FILENAME))</filename>
[safe:cat:capture] /usr/bin/cat <filename>((FILENAME))</filename>

# ============ DEBUG TEST PATTERNS ============
# Simple pattern for testing captures with quantifiers:
# a<b>b+</b>c<d>d*</d> should match:
#   - "abbbcd" -> capture "bbb" (pos 1-4) and "d" (pos 6)
#   - "abc" -> capture "b" (pos 1-2)
[debug:simple] a<b>b+</b>c<d>d*</d>

# ============ TEST PATTERNS FOR CAPTURES ============
# Simple capture pattern for testing
[safe:capture:test] cat \x3c<filename>((FILENAME))\x3e

# Note: 'find' commands are NOT in the DFA because they require detailed
# argument validation to detect dangerous options like -exec, -delete, etc.
# All 'find' commands go to the server where rofind module does full validation.

# ============ TEST PATTERNS FOR + QUANTIFIER ============
# Fragment definitions for + quantifier testing
[fragment:TEST::B] b
[fragment:TEST::Y] y
[fragment:TEST::TWO] 2

# Test patterns with + quantifier
# a((b))+ should match "ab", "abb", "abbb" but NOT "ac" or "abx"
[safe:quantifier:test] a((TEST::B))+
# x((y))+ should match "xy", "xyy", "xyyy" but NOT "xz"
[safe:quantifier:test] x((TEST::Y))+
# 1((2))+ should match "12", "12345" but NOT "13"
[safe:quantifier:test] 1((TEST::TWO))+
# ============ PLUS QUANTIFIER TEST PATTERNS ============
# Test patterns for + quantifier (one or more)
# These patterns test the fragment-based + quantifier

[fragment:b] b

# ============ FRAGMENTS FOR QUANTIFIER TEST PATTERNS ============
# Each pattern namespace needs its own fragment definitions

# Global fragments (fallback)
[fragment:b] b

[fragment:d] d

[fragment:z] z
